function upper_case(str)
{
regexp = /*[A-Z]/;
if (regexp.tst(str))
{
console.log("String's first character is uppercase");
}
else
{
console.log("String's first character is not uppercase");
}
}
upper_case('ABCD');
upper_case('abcd');
